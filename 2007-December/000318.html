+import "LangPrelude.prg"
+
+plus:: Nat ~> Nat ~> Nat
+{plus Z m} = m
+{plus (S n) m} = S {plus n m}
+
+prop Divsbl :: Nat ~> Nat ~> * where
+  Ground :: Divsbl 0t b
+  Leap :: Divsbl a b -> Divsbl {plus a b} b
+ deriving Nat(d)
+
+data Layout :: Nat ~> Nat ~> * where
+  Int :: Divsbl n 4t => Layout n (4+n)t
+  Pad :: Nat' p -> Layout n {plus n p}
