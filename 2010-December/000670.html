+import "Thrist.omg"
+
+data Count :: Nat ~> Nat ~> * where
+  Incr :: Nat' n -> Count n u -- (1+n)t
+
+shift :: Thrist Count n a -> Thrist Count (1+n)t b -- (1+a)t
+shift []l = []l
+shift [Incr n; r]l = [Incr (1+n)v; shift r]l
+
+a = [Incr 0v; lazy (shift a)]l
