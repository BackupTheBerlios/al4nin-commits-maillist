+kind Player = P1 | P2
+kind Serve = First | Second
+kind Team = T Serve Player Nat Player
+
+data Badminton :: Team ~> Team ~> *0 where
+  Start :: Badminton (T Second P1 #0 P2) (T First P1 #0 P2)
+  Point :: Badminton (T a p n q) t ->
+  	   	     Badminton (T a p (S n) q) t
+  Miss :: Badminton (T a p n q) t ->
+  	   	     Badminton (T a q n p) t
